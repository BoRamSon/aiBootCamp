			ORDER BY join_yyyy, player_name     -- 정렬이 안 먹는 문제가 있습니다.

UNION ALL

🟩 CASE¶

if 느낌에서 직관적으로 된 느낌으로 볼 수 있습니다.

🟢 Simple CASE¶

  • 특정 값(value) 을 기준으로 비교할 때 사용
CASE 컬럼명
  WHEN 값1 THEN 결과1
  WHEN 값2 THEN 결과2
  ELSE 기본값
END

SELECT 
  player_name,
  team_id,
  CASE team_id
    WHEN 'k01' THEN '기아 타이거즈'
    WHEN 'k02' THEN 'LG 트윈스'
    ELSE '기타 팀'
  END AS team_name
FROM player;

🟢 Searched CASE¶

  • 조건식(비교식) 을 직접 쓸 수 있음
CASE
  WHEN 조건1 THEN 결과1
  WHEN 조건2 THEN 결과2
  ELSE 기본값
END

SELECT 
  player_name,
  join_yyyy,
  CASE 
    WHEN join_yyyy < 2020 THEN '베테랑'
    WHEN join_yyyy >= 2020 AND join_yyyy <= 2023 THEN '중견'
    ELSE '신입'
  END AS 경력구분
FROM player;

🟩 CREATE & INSERT & DROP¶

분류 의미 주요 명령어 특징
DDL 데이터 정의어 CREATE, ALTER, DROP, TRUNCATE, RENAME 테이블 구조 변경, 자동 COMMIT
DML 데이터 조작어 SELECT, INSERT, UPDATE, DELETE 데이터 다룸, 트랜잭션 가능
DCL 데이터 제어어 GRANT, REVOKE 권한 관리

🟢 DDL (Data Definition Language)¶

데이터베이스의 구조(스키마)를 정의하거나 변경할 때 사용

항목 내용
주요 명령어 CREATE, ALTER, DROP, TRUNCATE, RENAME
특징 - 자동으로 COMMIT 됨 (ROLLBACK 불가능)
- 테이블/DB 구조 자체를 변경

예시¶

CREATE TABLE player (
    player_id INT PRIMARY KEY,
    name VARCHAR(50),
    height INT
);

ALTER TABLE player ADD COLUMN weight INT;
DROP TABLE player;

			writer int,          -- tb_member와 FOREIGN key로 묶음
			regdate datetime
);

-- 만들어졌는지 확인해봅시다
SHOW tables;


-- !!! FOREIGN KEY 추가하기     
-- tb_gestbook이 tb_member를 참조하는 구조이다.
ALTER TABLE tb_guestbook                             -- 테이블 'tb_guestbook'에 아래 내용을 추가하겠다
ADD CONSTRAINT fk_guestbook_member  -- 외래 키 제약 조건의 이름을 'fk_guestbook_member'라고 지정한다
FOREIGN KEY (writer)                                   -- 외래 키로 지정할 컬럼은 'writter'이다 (tb_guestbook 테이블 안의 컬럼)
REFERENCES tb_member(member_id);   -- 이 외래 키는 'tb_member' 테이블의 'member_id' 컬럼을 참조한다
